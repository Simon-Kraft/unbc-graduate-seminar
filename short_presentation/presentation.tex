\begin{frame}{Self-Attention}
    
    \begin{itemize}
        {\large 
        \item For sequence of input tokens $X = (x_1, \ldots, x_n) \in \mathbb{R}^{n \times d}$ compute self-attention with:
        \vspace{0.5em}
        \[
            \text{Attention}(Q,K,V) = \text{softmax}\Big(\frac{Q K^\top}{\sqrt{d}}\Big) V
        \]
        \item where the matrices $Q$, $K$, and $V$ are different linear projections of the input tokens
        }
    \end{itemize}

    \vspace{1.5em}

    \begin{block}{\large Result}
        {\large After this operation, each token captures \textbf{contextual information from all other tokens}.
        }
    \end{block}
\end{frame}

\begin{frame}{Problem Statement}
    \begin{block}{\large Token Processing }
        \begin{itemize}
            {\large
            \item Self-attention operation is executed in each layer
            \vspace{0.25em}
            \item Across layers, information from different tokens gets increasingly mixed
            }
        \end{itemize}

        \vspace{1.0em}

        \large \textbf{Research Question}: \textit{How can we quantify this information flow across multiple layers?}
    \end{block}
\end{frame}

\begin{frame}{Idea}
    \large Model information flow in the network with a \textbf{Directed Acyclic Graph (DAG)}:
    \begin{itemize}
        {\large 
        \item \textbf{Nodes}: Input tokens \& hidden embeddings
        \item \textbf{Edges}: Attentions from nodes in each layer to those in previous layer
        }
    \end{itemize}

    \vspace{1.5em}

    \begin{block}{\large Verb Number Prediction Task}
        \large \textit{Predicting singularity or plurality of a verb of a sentence, when the input is the sentence up to the verb position}
    \end{block}
\end{frame}
