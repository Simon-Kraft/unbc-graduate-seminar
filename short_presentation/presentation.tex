\begin{frame}{Self-Attention}
    
    \begin{itemize}
        {\large 
        \item For sequence of input tokens $X = (x_1, \ldots, x_n) \in \mathbb{R}^{n \times d}$ compute self-attention with:
        \vspace{0.5em}
        \[
            \text{Attention}(Q,K,V) = \text{softmax}\Big(\frac{Q K^\top}{\sqrt{d}}\Big) V
        \]

        \vspace{0.5em}

        where the matrices $Q$, $K$, and $V$ are different linear projections of the input tokens $X$
        \vspace{1.0em}
        \item $\text{softmax}\Big(\frac{Q K^\top}{\sqrt{d}}\Big)$ regarded as \textbf{attention weights}
        } \textcolor{red}{Explain linear projections and row normalization}
    \end{itemize}
    \vspace{1.5em}

    \begin{block}{\large Result}
        {\large After this operation, each token captures \textbf{contextual information from all other tokens}.
        }
    \end{block}
\end{frame}

\begin{frame}{Problem Statement}
    \begin{block}{\large Token Processing }
        \begin{itemize}
            {\large
            \item Self-attention operation is executed in \textbf{each layer}
            \vspace{0.25em}
            \item Across layers, \textbf{information} from different tokens gets increasingly \textbf{mixed}
            }
        \end{itemize}

        % \vspace{1.0em}

        % \large \textbf{Research Question}: \textit{How can we quantify this information flow across multiple layers?}
    \end{block}

    \vspace{1.5em}

    \begin{block}{\large Research Question}
        \large \textit{How can we quantify this information flow across multiple layers?}
    \end{block}
\end{frame}

\begin{frame}{Idea}
    \large Model information flow in the network with a \textbf{Directed Acyclic Graph (DAG)}:
    \begin{itemize}
        {\large 
        \item \textbf{Nodes}: Input tokens \& hidden embeddings
        \item \textbf{Edges}: Attention weights from nodes in each layer to those in previous layer
        }
    \end{itemize}

    \vspace{1.5em}

    \begin{block}{\large Verb Number Prediction Task}
        \large \textit{Predicting singularity or plurality of a verb of a sentence, when the input is the sentence up to the verb position}
    \end{block}
\end{frame}

\begin{frame}{Methods}
    \begin{block}{\large Attention Rollout}
        \begin{itemize}
            {\large
            \item Recursively multiply \textbf{attention weight matrices} across transformer layers:
            \[
                \tilde{A}^{(l)} =  \left(A^{(l)} +  \mathbb{I} \right) \cdot \tilde{A}^{(l-1)},
                \quad \tilde{A}^{(0)} = \mathbb{I}
            \]
    
            \vspace{0.5em}
    
            where $A^{(l)} = $ are the attention weights from layer $l$
            
            \vspace{0.5em}

            \item \textbf{Row-normalization} after each multiplication
            \vspace{0.25em}
            \item Whole operation in $\mathcal{O}(d \times n^2)$
            }
        \end{itemize}
        \vspace{0.5em}
        \hrule
        \vspace{0.1em}
        {\scriptsize \textbf{Notation}: $d$ is model depth, $n$ is sequence length.}
    \end{block}
\end{frame}

\begin{frame}{Methods}
    \begin{block}{\large Attention Flow}
        \begin{itemize}
            {\large
            \setlength{\itemsep}{0.5em}
            \item Attention graph as a \textbf{flow network}
            
            \item Capacities of edges are attention weights
            \item \textbf{Maximum flow algorithm} to compute attention flow from \textbf{one} node in output layer to \textbf{one} node in input layer
            \item One computation in $\mathcal{O}(d^2 \times n^4)$
            }
        \end{itemize}
    \end{block}
\end{frame}
