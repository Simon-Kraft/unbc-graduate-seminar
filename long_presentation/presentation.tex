\begin{frame}{Federated Learning Training Protocol}
    {\large \textbf{For each training round:}}
    % Added <+-> to reveal list items one by one
    \begin{enumerate}[<+->]
        \item \textbf{Local Training:} Clients train for one epoch with their private datasets
        \item \textbf{Compute Loss:} Clients calculate loss gradient 
        \item \textbf{Upload:} Gradients (updates) sent to global server
        \item \textbf{Aggregation:} Server aggregates updates (e.g. averaging) and updates global model
        \item \textbf{Distribution:} Updated global model weights sent to clients
    \end{enumerate}
    \vspace{1.5em}
    % Block appears only after the list is finished
    \begin{block}{Assumption}<6>
        \centering
        \textbf{Only loss gradients shared $\Rightarrow$ privacy-preserving}
    \end{block}
\end{frame}

\begin{frame}{First Solution: Differential Privacy}
    \begin{block}{Differential Privacy (DP)}
        Mathematical framework for releasing statistical information about datasets while protecting the privacy of individual data subjects.\footnote{\url{https://en.wikipedia.org/wiki/Differential_privacy}} 
    \end{block}
    \pause
    \textbf{In Federated Learning}
    \vspace{-0.5em}
    \begin{itemize}[<+->]
        \item Usually achieved by adding Gaussian noise to gradients
        \item Requires careful calculations to compute the amount of noise that needs to be added
    \end{itemize}
    \vspace{1em}
    \pause
    \textbf{\textcolor{red}{Problem:}} Accuracy reduction over 20\% and longer training time
\end{frame}

\begin{frame}{InstaHide: Method}
    \textbf{Goal:} Ensure privacy in federated learning without sacrificing accuracy.
    
    \vspace{1em}

    % 1. The block appears on slide 2
    \begin{block}{InstaHide}<2->
        Two step procedure to encrypt a private training image
        
        % 2. The list items start revealing one-by-one starting from slide 3
        \begin{enumerate}[<+(2)- | handout:0>]
            \item \textbf{\textit{Mixup} augmentation:}
            Take linear combination of private image with $k-1$ randomly chosen images from both private and public datasets~\cite{zhang2018mixupempiricalriskminimization}
            
            \vspace{0.75em}
            
            \item \textbf{Pixel-wise sign-flipping:} 
            Apply random sign-flipping mask ($+1$ or $-1$) to composite image to destroy visual structure.
        \end{enumerate}
    \end{block}
\end{frame}

\begin{frame}{InstaHide: Training and Inference}
    % First block appears on slide 2
    \begin{block}{Training}<2->
        Authors present two schemes for encryption
        \begin{itemize}[<+(2)- | handout:0>]
            \item \textbf{Inside-Dataset:} Mixing only with random images within the same private training dataset
            \item \textbf{Cross-Dataset:} Mixing with random images from private training dataset and public dataset
        \end{itemize}
    \end{block}

    \vspace{1em}

    % Second block appears after the first list is finished (slide 5)
    \begin{block}{Inference}<5->
        Both schemes applied during inference
        \begin{itemize}[<+(4)- | handout:0>]
            \item Test image is encrypted \textbf{multiple times} (e.g. 10)
            \item Average predictions
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{InstaHide: Dynamic Encryption}
    \textbf{Note:} Every private image is re-encrypted every epoch
    
    % The list starts revealing one-by-one starting from slide 2
    \begin{itemize}[<+(1)- | handout:0>]
        \item Different blend of private and public images 
        \item Different mixing ratio ($\lambda$'s)
        \item Different sign-flipping mask
    \end{itemize}
    
    \vspace{0.75em}

    % The block appears after the 3 items (slide 5)
    \begin{block}{Result}<5->
        \textit{One-time encryption key} that is never reused for another encryption.
    \end{block}
\end{frame}

\begin{frame}{Carlini et al.'s Attack}
    The publication \citetitle{carlini2021privatelearningpossibleinstance}~\cite{carlini2021privatelearningpossibleinstance} successfully \textbf{broke encryption} scheme of \textit{InstaHide}. 
    \vspace{1em}
    \pause
    \begin{block}{Approach}
        \begin{enumerate}[<+->]
            \item Train neural network for similarity annotation
            \vspace{0.25em}
            \item Run combinatorial clustering algorithm to cluster based on private images (\textbf{cubic} time in size of private dataset)
            \vspace{0.25em}
            \item A regression algorithm solves for the original pixels
        \end{enumerate}
    \end{block}
\end{frame}

\begin{frame}{Conclusion}
    \begin{block}{Utility}
        \begin{itemize}[<+->]
            \item \textit{InstaHide} provides light-weight encryption method for distributed learning
            \item Small accuracy drop compared to vanilla training
            \item Higher inference time due to prediction averaging
        \end{itemize}
    \end{block}
    \vspace{1em}
    \pause
    \begin{block}{Security}
        \begin{itemize}[<+->]
            \item One-time encryption key in every epoch
            \item Effective defense against \citetitle{zhu2019deepleakagegradients}
            \item Attackers found $\mathcal{O}(N^3)$ algorithm to break encryption
        \end{itemize}
    \end{block}
\end{frame}
